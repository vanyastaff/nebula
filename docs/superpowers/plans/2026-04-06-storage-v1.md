# Storage v1 — PgExecutionRepo Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the Postgres execution repository so the engine can persist execution state, per-node outputs, journal entries, and idempotency keys — surviving crashes.

**Architecture:** Extend the existing `ExecutionRepo` trait with methods for node outputs, listing, and idempotency. Implement `PgExecutionRepo` following the same patterns as `PgWorkflowRepo` (pool-based, CAS via version column, `sqlx::query` with bind params). 4 new Postgres tables. JSONB for all data columns (MessagePack optimization deferred to Phase 13).

**Tech Stack:** Rust 1.94, sqlx 0.8 (Postgres), `cargo test --workspace`, `cargo clippy --workspace -- -D warnings`. Postgres via `task db:up` (Docker Compose). sqlx offline mode for CI (`task db:prepare`).

async fn create(
    &self,
    id: ExecutionId,
    workflow_id: WorkflowId,
    state: serde_json::Value,
) -> Result<(), ExecutionRepoError> {
    let mut states = self.state.write().await;
    if states.contains_key(&id) {
        return Err(ExecutionRepoError::Conflict {
            entity: "execution".into(),
            id: id.to_string(),
            expected_version: 0,
            actual_version: 1,
        });
    }
    states.insert(id, (1, state));
    self.workflows.write().await.insert(id, workflow_id);
    Ok(())
}

- [ ] **Step 3: Implement node output methods**

```rust
async fn save_node_output(
    &self,
    execution_id: ExecutionId,
    node_id: NodeId,
    attempt: u32,
    output: serde_json::Value,
) -> Result<(), ExecutionRepoError> {
    self.node_outputs.write().await
        .insert((execution_id, node_id, attempt), output);
    Ok(())
}

async fn load_node_output(
    &self,
    execution_id: ExecutionId,
    node_id: NodeId,
) -> Result<Option<serde_json::Value>, ExecutionRepoError> {
    let outputs = self.node_outputs.read().await;
    // Find the highest attempt for this execution+node
    let result = outputs.iter()
        .filter(|((eid, nid, _), _)| *eid == execution_id && *nid == node_id)
        .max_by_key(|((_, _, attempt), _)| *attempt)
        .map(|(_, v)| v.clone());
    Ok(result)
}

async fn load_all_outputs(
    &self,
    execution_id: ExecutionId,
) -> Result<HashMap<NodeId, serde_json::Value>, ExecutionRepoError> {
    let outputs = self.node_outputs.read().await;
    let mut result: HashMap<NodeId, (u32, serde_json::Value)> = HashMap::new();
    for ((eid, nid, attempt), value) in outputs.iter() {
        if *eid == execution_id {
            let entry = result.entry(*nid).or_insert((0, serde_json::Value::Null));
            if *attempt > entry.0 {
                *entry = (*attempt, value.clone());
            }
        }
    }
    Ok(result.into_iter().map(|(k, (_, v))| (k, v)).collect())
}
```

```rust
async fn list_running(&self) -> Result<Vec<ExecutionId>, ExecutionRepoError> {
    let states = self.state.read().await;
    let mut running = Vec::new();
    for (id, (_, state_json)) in states.iter() {
        if let Some(status) = state_json.get("status").and_then(|s| s.as_str()) {
            if matches!(status, "created" | "running" | "paused" | "cancelling") {
                running.push(*id);
            }
        }
    }
    Ok(running)
}

async fn count(
    &self,
    workflow_id: Option<WorkflowId>,
) -> Result<u64, ExecutionRepoError> {
    match workflow_id {
        None => Ok(self.state.read().await.len() as u64),
        Some(wid) => {
            let workflows = self.workflows.read().await;
            Ok(workflows.values().filter(|v| **v == wid).count() as u64)
        }
    }
}

async fn check_idempotency(&self, key: &str) -> Result<bool, ExecutionRepoError> {
    Ok(self.idempotency.read().await.contains(key))
}

async fn mark_idempotent(
    &self,
    key: &str,
    _execution_id: ExecutionId,
    _node_id: NodeId,
) -> Result<(), ExecutionRepoError> {
    self.idempotency.write().await.insert(key.to_string());
    Ok(())
}

//! PostgreSQL-backed execution repository.

use std::collections::HashMap;
use std::time::Duration;

use async_trait::async_trait;
use sqlx::types::Json;
use sqlx::{Pool, Postgres};

use nebula_core::{ExecutionId, NodeId, WorkflowId};

use crate::execution_repo::{ExecutionRepo, ExecutionRepoError};

/// PostgreSQL-backed implementation of [`ExecutionRepo`].
///
/// Uses CAS (compare-and-swap) via a `version` column for optimistic concurrency.
/// Follows the same patterns as [`PgWorkflowRepo`](super::PgWorkflowRepo).
#[derive(Clone)]
pub struct PgExecutionRepo {
    pool: Pool<Postgres>,
}

impl PgExecutionRepo {
    /// Create a new repository backed by the given connection pool.
    pub fn new(pool: Pool<Postgres>) -> Self {
        Self { pool }
    }

    /// Helper: convert nebula ExecutionId to sqlx UUID.
    fn to_uuid(id: ExecutionId) -> sqlx::types::Uuid {
        sqlx::types::Uuid::from_bytes(*id.get().as_bytes())
    }

    /// Helper: convert nebula NodeId to sqlx UUID.
    fn node_to_uuid(id: NodeId) -> sqlx::types::Uuid {
        sqlx::types::Uuid::from_bytes(*id.get().as_bytes())
    }

    /// Helper: convert nebula WorkflowId to sqlx UUID.
    fn workflow_to_uuid(id: WorkflowId) -> sqlx::types::Uuid {
        sqlx::types::Uuid::from_bytes(*id.get().as_bytes())
    }

    /// Helper: convert sqlx UUID to nebula ExecutionId.
    fn from_uuid(uuid: sqlx::types::Uuid) -> ExecutionId {
        ExecutionId::from_bytes(*uuid.as_bytes())
    }

    /// Helper: convert sqlx UUID to nebula NodeId.
    fn node_from_uuid(uuid: sqlx::types::Uuid) -> NodeId {
        NodeId::from_bytes(*uuid.as_bytes())
    }
}
```

```rust
#[async_trait]
impl ExecutionRepo for PgExecutionRepo {
    async fn create(
        &self,
        id: ExecutionId,
        workflow_id: WorkflowId,
        state: serde_json::Value,
    ) -> Result<(), ExecutionRepoError> {
        let uuid = Self::to_uuid(id);
        let wf_uuid = Self::workflow_to_uuid(workflow_id);
        let json_state = Json(&state);

        match sqlx::query(
            "INSERT INTO executions (id, workflow_id, version, state) VALUES ($1, $2, 1, $3)",
        )
        .bind(uuid)
        .bind(wf_uuid)
        .bind(json_state)
        .execute(&self.pool)
        .await
        {
            Ok(_) => Ok(()),
            Err(sqlx::Error::Database(db_err))
                if db_err.code().as_deref() == Some("23505") =>
            {
                Err(ExecutionRepoError::Conflict {
                    entity: "execution".into(),
                    id: id.to_string(),
                    expected_version: 0,
                    actual_version: 1,
                })
            }
            Err(err) => Err(ExecutionRepoError::Connection(err.to_string())),
        }
    }

    async fn get_state(
        &self,
        id: ExecutionId,
    ) -> Result<Option<(u64, serde_json::Value)>, ExecutionRepoError> {
        let uuid = Self::to_uuid(id);

        let row = sqlx::query_as::<_, (i64, Json<serde_json::Value>)>(
            "SELECT version, state FROM executions WHERE id = $1",
        )
        .bind(uuid)
        .fetch_optional(&self.pool)
        .await
        .map_err(|e| ExecutionRepoError::Connection(e.to_string()))?;

        Ok(row.map(|(v, Json(s))| (v as u64, s)))
    }
```
